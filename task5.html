// Constants & DOM Elements const UNIT_PRICE = 100; const quantityInput = document.getElementById('quantity'); const payableDisplay = document.getElementById('payable'); const addToCartBtn = document.getElementById('addToCartBtn'); /** * Calculates and updates the total payable amount */ function updatePayable() { const quantity = parseInt(quantityInput.value) || 0; const total = UNIT_PRICE * quantity; // Update the visual box text payableDisplay.textContent = `$${total}`; } // Listen for quantity changes to update the price instantly quantityInput.addEventListener('input', updatePayable); // Handle the "Add to Cart" button click addToCartBtn.addEventListener('click', () => { const quantity = quantityInput.value; const total = UNIT_PRICE * quantity; if (quantity > 0) { alert(`Successfully added ${quantity} items to your cart! Total: $${total}`); } else { alert('Please select a valid quantity.'); } });
